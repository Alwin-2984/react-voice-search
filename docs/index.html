Basic Usage:

import React from "react";
import VoiceSearch from "react-voice-search";

function App() {
  const handleSearch = (transcript) => {
    console.log("User said:", transcript);
  };

  return (
    <VoiceSearch 
      handleSearch={handleSearch}
      placeholder="Click mic and speak..."
    />
  );
}
